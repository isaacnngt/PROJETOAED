void printar_matriz(int qtd,Restaurante grafo[][10])//inicializando matriz/grafo
{
    for(int l = 0; l < qtd; l++)
    {
        for(int c = 0; c < qtd; c++)
        {
            printf("(%d ", grafo[l][c].peso);
            printf("%s) ", grafo[l][c].nome);
        }
        printf("\n");
    }
    printf("\n\n");

    for(int l = 0; l < qtd; l++)

        for(int c = 0; c < qtd; c++)

            printf("%s ", grafo[l][c].nome);

        printf("\n");
    }
    printf("\n\n");
}
